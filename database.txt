CREATE DATABASE db_buku;
USE db_buku;

CREATE TABLE buku (
    id INT AUTO_INCREMENT PRIMARY KEY,
    judul VARCHAR(100),
    penulis VARCHAR(100),
    harga INT,
    gambar VARCHAR(100)
);
